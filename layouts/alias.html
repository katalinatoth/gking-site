{{- /* Custom alias template. Normal Hugo aliases redirect to the target page's canonical URL. We additionally honour a fragment in the FIRST alias entry so that (for example) /advice/ can land on /teaching/#advice. */ -}} {{- $target := .Permalink -}} {{- with .Page -}} {{- $aliases := .Aliases | default slice -}} {{- range $a := $aliases -}} {{- $s := string $a -}} {{- if (strings.Contains $s "#") -}} {{- $target = printf "%s#%s" $.Permalink (index (split $s "#") 1) -}} {{- end -}} {{- end -}} {{- end -}} {{ $target }}
