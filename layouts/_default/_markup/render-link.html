{{- /* Link render hook. • Absolute-path links starting with "/" (e.g. `/publication/slug/`, `/files/foo.pdf`) are made baseURL-aware so they keep working under the site's /gking-site/ subpath on GitHub Pages (and under any future baseURL). • `http://` / `https://` / `mailto:` / `tel:` links that leave the site get `target="_blank" rel="noopener"` so they open in a new tab. • Empty anchors (`#…`) and already-prefixed paths are left alone. */ -}} {{- $dest := .Destination -}} {{- $text := .Text -}} {{- $title := .Title -}} {{- $external := false -}} {{- $internal := false -}} {{- if or (strings.HasPrefix $dest "http://") (strings.HasPrefix $dest "https://") -}} {{- /* Absolute URL: treat as external (even if it happens to resolve to this site). */ -}} {{- $external = true -}} {{- else if strings.HasPrefix $dest "/" -}} {{- /* Site-absolute path: route through relURL so the /gking-site/ prefix is applied. */ -}} {{- $dest = relURL (strings.TrimPrefix "/" $dest) -}} {{- $internal = true -}} {{- end -}} {{- $attrs := "" -}} {{- if $external -}} {{- $attrs = " target=\"_blank\" rel=\"noopener\"" -}} {{- end -}} {{- if $title -}} {{ $text | safeHTML }} {{- else -}} {{ $text | safeHTML }} {{- end -}}
