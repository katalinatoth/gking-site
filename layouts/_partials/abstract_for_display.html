{{- /* When Params.dataverse_url is set, the abstract often still ends with a run-on sentence like "Replication data at the Harvard Dataverse:https://…". That duplicates the Dataverse callout and See also entry — strip it here for list cards and single-page abstract blocks. */ -}} {{- $t := .Params.abstract | default "" -}} {{- if and $t .Params.dataverse_url -}} {{- $t = replaceRE `(?i)\s*\.?\s*replication data at the harvard dataverse:\s*https?://\S+` "" $t -}} {{- $t = trim $t " \n\t." -}} {{- end -}} {{- return $t -}}
